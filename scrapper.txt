    async function toggleBookmark(e, code, title) {
        e.preventDefault(); 
        e.stopPropagation();

        const baseCode = code.substring(0, 8);
        const existingIndex = rakData.findIndex(item => item.kode_modul === baseCode);

        try {
            const formData = new FormData();
            formData.append('ajax_action', 'toggle_rak');
            formData.append('csrf_token', csrfToken);
            formData.append('modul_code', baseCode);
            formData.append('modul_title', title);

            const response = await fetch('ajax_rak.php', { method: 'POST', body: formData });
            const result = await response.json();

            if (result.status === 'full') {
                Swal.fire({ icon: 'warning', title: 'Rak Pribadi Penuh!', text: 'Kapasitas maksimal 10 modul. Silakan hapus modul lama terlebih dahulu.', confirmButtonColor: '#005baa', borderRadius: '24px' });
            } else if (result.status === 'spam') {
                Swal.fire({ icon: 'error', title: 'Terlalu Cepat!', text: result.message, confirmButtonColor: '#dc2626', borderRadius: '24px' });
            } else if (result.status === 'success') {
                rakData = result.data; 
                renderRakPribadi();
                
                if (existingIndex > -1) {
                    Swal.fire({ toast:true, position:'top-end', icon:'success', title:'Dihapus dari Rak Pribadi', showConfirmButton:false, timer:2000 });
                } else {
                    Swal.fire({ toast:true, position:'top-end', icon:'success', title:'Tersimpan di Rak Pribadi', showConfirmButton:false, timer:2000 });
                }
                
                if(document.getElementById('searchInput').value.length >= 3) {
                    filterModul(); 
                }
            } else {
                Swal.fire({ toast:true, position:'top-end', icon:'error', title: result.message || 'Terjadi kesalahan', showConfirmButton:false, timer:2000 });
            }
        } catch (error) {
            Swal.fire({ toast:true, position:'top-end', icon:'error', title:'Gagal menyambung ke server', showConfirmButton:false, timer:2000 });
        }

    function getHighestEditionFolder(baseCode) {
        let bestMatch = baseCode;
        let maxEdition = 1;
        const regex = new RegExp('^' + baseCode + '(\\d{2})?$');

        for(let i = 0; i < availableFolders.length; i++) {
            let folder = availableFolders[i];
            let match = folder.match(regex);
            if(match) {
                let edition = match[1] ? parseInt(match[1], 10) : 1;
                if(edition >= maxEdition) {
                    maxEdition = edition;
                    bestMatch = folder;
                }
            }
        }
        return { folder: bestMatch, edition: maxEdition };
